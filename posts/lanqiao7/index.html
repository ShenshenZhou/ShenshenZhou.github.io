bool check()
{
	bool b1 = (a[0] + a[1] == a[2]);
	bool b2 = (a[3] - a[4] == a[5]);
	bool b3 = (a[6] * a[7] == a[8]);
	bool b4 = (fabs((a[9] * 1.0) / (a[10] * 1.0) - a[11] * 1.0) <= 0.00000000000001);
	if (b1 && b2 && b3 && b4)
		return true;
	else
		return false;
}

// 最直接的搜索 参考链接优化
#include<iostream>
#include<cmath>

using namespace std;

int n;

int main() {
	cin >> n;
	int root = sqrt(n) + 1;
	for (int i = 0; i <= root; ++i) {
		for (int j = 0; j <= root; ++j) {
			for (int k = 0; k <= root; ++k) {
				for (int l = 0; l <= root; ++l) {
					if (pow(i, 2) + pow(j, 2) + pow(k, 2) + pow(l, 2) == n) {
						cout << i << " " << j << " " << k << " " << l << endl;
						return 0;
					}
					
				}

			}
		}
	}

	return 0;
}

#include<iostream>

using namespace std;

const int n = 1005;
string s;
int dp[n][n];  // dp[i][j]表示以i为起点 j为终点的字符串中的最大回文串

int main() {
	cin >> s;
	int len = s.size();
	for (int i = 0; i < len; ++i) {
		dp[i][i] = 1;  // 单个字符就是长度为1的回文串 初始化为1
	}
	for (int i = 2; i <= len; ++i) {
		// 枚举回文串的长度 从第二个到最后一个字符依次求最大回文串长度
		for (int l = 0; l + i - 1 < len; ++l) {
			int r = i + l - 1;
			if (s[l] == s[r])
				// 左边界==右边界 左边前移 右边后移 长度+2
				dp[l][r] = dp[l + 1][r - 1] + 2;
			else
				// 不相等 最大长度为左边前移一位 和 右边后移一位的最大者
				dp[l][r] = max(dp[l][r - 1], dp[l + 1][r]);
		}
	}

	cout << len - dp[0][len - 1];
	return 0;
}
