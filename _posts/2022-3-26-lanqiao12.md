[试题I：双向排序](https://blog.csdn.net/Prudento/article/details/123564856)

>暴力循环，vector容器，sort()排序，注意边界元素的值。

```c++
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

const int maxn = 1e6+1;

int main() {
	vector<int> vec(maxn);
	int n, m;
	cin >> n >> m;
	
	for (int i = 0; i < n; ++i) {
		vec[i] = 1 + i;
	}

	for (int i = 0; i < m; ++i) {
		int p, q;
		cin >> p >> q;
		if (!p) {
			sort(vec.begin(), vec.begin() + q, greater<int>());
		} else {
			// 在这里调试了半天 迭代器指向的位置要对
			sort(vec.begin() + q - 1, vec.begin() + n);
		}
	}

	for (int i = 0; i < n; ++i) {
		cout << vec[i] << " ";
	}

	return 0;
}
```
